    "def rp_challenge() -> bytes:\n",
    "    \"\"\"RP issues a fresh random challenge (WebAuthn rsp).\"\"\"\n",
    "    return secrets.token_bytes(32)\n",
    "\n",
    "def authenticator_sign(sk: SigningKey, challenge: bytes):\n",
    "    \"\"\"Authenticator signs challenge with its passkey \u2192 (vk, signature).\"\"\"\n",
    "    sig = sk.sign(challenge, hashfunc=hashlib.sha256)\n",
    "    return sk.get_verifying_key(), sig\n",
    "\n",
    "def cd_algorithm(vk, sig, challenge, V, V_prime_idx) -> str:\n",
    "    \"\"\"\n",
    "    Compromise Detection algorithm (Figure 5 of paper).\n",
    "    Returns: 'ACCEPT' | 'ALARM' | 'REJECT'\n",
    "    \"\"\"\n",
    "    # Step 1: check v \u2208 V\n",
    "    vk_bytes = vk.to_string()\n",
    "    matched_idx = next((i for i, vi in enumerate(V)\n",
    "                        if vi.to_string() == vk_bytes), None)\n",
    "    if matched_idx is None:\n",
    "        return 'REJECT'  # v \u2209 V\n",
    "\n",
    "    # Step 2: verify signature\n",
    "    try:\n",
    "        valid = V[matched_idx].verify(sig, challenge, hashfunc=hashlib.sha256)\n",
    "    except Exception:\n",
    "        return 'REJECT'\n",
    "\n",
    "    # Step 3: v \u2208 V' \u2192 breach alarm\n",
    "    if matched_idx in V_prime_idx:\n",
    "        return 'ALARM'\n",

    "    \"\"\"Simulate caat's success rate at guessing w_i* on the first try.\"\"\"\n",

    "def min_tdp_simulation(k_bar: int, alpha: float, m: int, n: int,\n",
    "                       n_trials: int = 3000) -> float:\n",
    "    \"\"\"\n",
    "    Monte-Carlo estimate of minimum TDP.\n",
    "    \n",
    "    Parameters\n",
    "    ----------\n",
    "    k_bar  : effective decoy count (perfectly flat W\u0304)\n",
    "    alpha  : fraction of active decoy verifiers\n",
    "    m      : number of unbreached RPs caat wants to compromise\n",
    "    n      : number of already-breached RPs caat observed\n",
    "    \"\"\"\n",
    "    detected_count = 0\n",
    "\n",
    "    for _ in range(n_trials):\n",
    "        # Index space: 0 = real, 1..k_bar = decoys\n",
    "        all_indices = list(range(k_bar + 1))\n",
    "\n",
    "        # caat observes n breaches: each reveals |V'| = ceil(\u03b1*k_bar) active decoys.\n",
    "        # caat can rule out those indices from its guess pool.\n",
    "        n_active = math.ceil(alpha * k_bar)\n",
    "        known_bad = set()\n",
    "        for _ in range(n):\n",
    "            # Each breach reveals a fresh random set of active decoy indices\n",
    "            # (excluding real=0)\n",
    "            decoy_pool = [i for i in all_indices if i != 0 and i not in known_bad]\n",
    "            batch = random.sample(decoy_pool, min(n_active, len(decoy_pool)))\n",
    "            known_bad.update(batch)\n",
    "\n",
    "        # Remaining candidates: W\u0304(n) = indices not known to be active decoys\n",
    "        W_bar_n = [i for i in all_indices if i not in known_bad]\n",
    "\n",
    "        # caat's optimal strategy: reuse the same guess if it wasn't detected,\n",
    "        # otherwise pick a new one.\n",
    "        # MDP best strategy: pick one w uniformly from W\u0304(n), reuse until detected.\n",
    "        guess = random.choice(W_bar_n)\n",
    "\n",
    "        # Simulate m RP attack attempts\n",
    "        detected = False\n",
    "        active_decoy_set = set(range(1, k_bar + 1))  # 0 = real\n",
    "        # Active decoys at each unbreached RP are a fresh \u03b1-fraction of decoys\n",
    "        for _ in range(m):\n",
    "            # Fresh active decoy assignment at this RP\n",
    "            active_here = set(random.sample(\n",
    "                list(range(1, k_bar + 1)), n_active))\n",
    "            if guess != 0 and guess in active_here:\n",
    "                detected = True\n",
    "                break\n",
    "            # Not detected: caat may reuse or switch\n",
    "            # Optimal: if guess==0 (real), always reuse (never detected)\n",
    "            # If guess is inactive decoy, keep using it (not detected yet)\n",
    "\n",
    "        if detected:\n",
    "            detected_count += 1\n",
    "\n",
    "    return detected_count / n_trials\n",
    "\n",
    "\n",
